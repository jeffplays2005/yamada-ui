## Theming

The `Status` is a [multi part component](/styled-system/theming/component-styles#multi-part-components).

:::note
If you want to change the style of the component, please check [here](/styled-system/theming/customize-theme#changing-the-style-of-components).
:::

```ts
export const Status: ComponentMultiStyle<"Status"> = {
  baseStyle: ({ colorScheme: bg = "info" }) => ({
    container: {
      alignItems: "center",
      display: "flex",
      gap: "sm",
    },
    indicator: {
      bg,
      flexShrink: 0,
      forcedColorAdjust: "none",
      rounded: "full",
    },
  }),

  sizes: {
    sm: {
      indicator: { boxSize: "2" },
      label: { fontSize: "sm" },
    },
    md: {
      indicator: { boxSize: "2.5" },
      label: { fontSize: "md" },
    },
    lg: {
      indicator: { boxSize: "3" },
      label: { fontSize: "lg" },
    },
  },

  defaultProps: {
    colorScheme: "info",
    size: "md",
  },
}
```
