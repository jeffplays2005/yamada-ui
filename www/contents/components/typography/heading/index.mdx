## Import

```ts
import { Heading } from "@yamada-ui/react"
```

## Usage

```tsx
<Heading>ギャルのパンティーおくれーーーっ！！！！！</Heading>
```

### Change the size

```tsx
<VStack>
  <For
    each={[
      { as: "h1", size: "4xl" },
      { as: "h2", size: "3xl" },
      { as: "h2", size: "2xl" },
      { as: "h3", size: "xl" },
      { as: "h3", size: "lg" },
      { as: "h4", size: "md" },
      { as: "h5", size: "sm" },
      { as: "h6", size: "xs" },
    ]}
  >
    {({ as, size }, index) => (
      <Heading key={index} as={as} size={size} isTruncated>
        ギャルのパンティーおくれーーーっ！！！！！
      </Heading>
    )}
  </For>
</VStack>
```

### Add a gradient

```tsx
<Heading
  w="full"
  size="2xl"
  bgGradient="linear(to-l, #7928CA, #FF0080)"
  bgClip="text"
  isTruncated
>
  クリリンのことか……クリリンのことかーーーっ！！！！！
</Heading>
```
