:::note status=error
`Divider` is planned to be removed in `v2.0`. Use [Separator](/components/layouts/separator) instead.
:::

## Import

```ts
import { Divider } from "@yamada-ui/react"
```

## Usage

```tsx
<Flex direction="column" gap="md">
  <For each={["solid", "dashed", "dotted"]}>
    {(variant, index) => <Divider key={index} variant={variant} />}
  </For>
</Flex>
```

### Draw a vertical dividing line

```tsx
<Flex h="sm" gap="md">
  <For each={["solid", "dashed", "dotted"]}>
    {(variant, index) => (
      <Divider key={index} orientation="vertical" variant={variant} />
    )}
  </For>
</Flex>
```

:::note status=warning
When making the dividing line vertical, you need to set the height to the parent element or `Divider`.
:::
